class EnvironmentModel:
    """
    Physical environment model for the mushroom cultivation chamber.

    Responsibilities
    ----------------
    • Temperature Dynamics
    • Humidity Dynamics 
    • CO₂ Dynamics 
    • Moisture Dynamics 

    Mathematical Models
    -------------------
    • First-order thermal model
    • First-order humidity model
    • CO₂ mass balance
    • Moisture balance
    """

    def __init__(self):

        # ---------------------------------------------------------------------
        # Temperature Model Parameters
        # ---------------------------------------------------------------------

        self.ka = Config.KA
        self.kc = Config.KC
        self.kh = Config.KH
        self.kr = Config.KR

        # ---------------------------------------------------------------------
        # Humidity Model Parameters
        # ---------------------------------------------------------------------

        self.kf = Config.KF
        self.kv = Config.KV
        self.kt = Config.KT

        # ---------------------------------------------------------------------
        # Moisture Model Parameters
        # ---------------------------------------------------------------------

        self.ks = Config.KS
        self.kg = Config.KG

        # ---------------------------------------------------------------------
        # Evaporation Parameters
        # ---------------------------------------------------------------------

        self.alpha = Config.ALPHA
        self.beta = Config.BETA

        # ---------------------------------------------------------------------
        # Ambient Conditions
        # ---------------------------------------------------------------------

        self.ambient_temperature = Config.AMBIENT_TEMPERATURE

    # =========================================================================
    # PRIVATE HELPER METHODS
    # =========================================================================

    def _clamp(
        self,
        value: float,
        minimum: float,
        maximum: float
    ) -> float:
        """
        Restrict a value within a specified range.
        """

        return max(minimum, min(value, maximum))

    # =========================================================================
    # EVAPORATION MODEL
    # =========================================================================

    def evaporation_rate(
        self,
        temperature: float,
        humidity: float
    ) -> float:
        """
        Calculates the evaporation rate.

        Engineering Model

        ke = α(T − 20) + β(100 − H)

        Parameters
        ----------
        temperature : float
            Chamber temperature (°C)

        humidity : float
            Relative humidity (%)

        Returns
        -------
        float
            Evaporation coefficient
        """

        evaporation = (

            self.alpha *

            (temperature - 20.0)

            +

            self.beta *

            (100.0 - humidity)

        )

        evaporation = max(evaporation, 0.0)

        return evaporation

    # =========================================================================
    # TEMPERATURE MODEL
    # =========================================================================

    def update_temperature(
        self,
        state: TwinState
    ) -> None:
        """
        First-order chamber thermal dynamics.

        Tk+1 = Tk + Δt[
            ka(Tamb - Tk)
            - kc(Uc)
            + kh(Uh)
            + kr(Qr)
        ]
        """

        # ---------------------------------------------------------------------
        # Ambient Heat Transfer
        # ---------------------------------------------------------------------

        ambient_heat = (

            self.ka *

            (

                self.ambient_temperature

                -

                state.temperature

            )

        )

        # ---------------------------------------------------------------------
        # Cooling Pad Effect
        # ---------------------------------------------------------------------

        cooling_effect = 0.0

        if state.cooling_pad:

            cooling_effect = self.kc

        # ---------------------------------------------------------------------
        # Heater Effect
        # ---------------------------------------------------------------------

        heating_effect = 0.0

        if state.heater:

            heating_effect = self.kh

        # ---------------------------------------------------------------------
        # Respiration Heat
        # ---------------------------------------------------------------------

        respiration_effect = (

            self.kr *

            state.respiration_heat

        )

        # ---------------------------------------------------------------------
        # Temperature Change
        # ---------------------------------------------------------------------

        delta_temperature = (

            ambient_heat

            -

            cooling_effect

            +

            heating_effect

            +

            respiration_effect

        )

        # ---------------------------------------------------------------------
        # Euler Integration
        # ---------------------------------------------------------------------

        state.temperature += (

            Config.TIME_STEP *

            delta_temperature

        )

        # ---------------------------------------------------------------------
        # Physical Limits
        # ---------------------------------------------------------------------

        state.temperature = self._clamp(

            state.temperature,

            0.0,

            50.0

        )

    # =========================================================================
    # HUMIDITY MODEL
    # =========================================================================

    def update_humidity(
        self,
        state: TwinState
    ) -> None:
        """
        Humidity Dynamics

        H(k+1) = H(k) + Δt[
            kf(Uf)
            - kv(Ufan)
            + kt(W)
            - ke(T,H)
        ]
        """

        # -------------------------------------------------------------
        # Evaporation Loss
        # -------------------------------------------------------------

        evaporation_loss = self.evaporation_rate(
            state.temperature,
            state.humidity
        )

        # -------------------------------------------------------------
        # Fogger Contribution
        # -------------------------------------------------------------

        fogger_gain = 0.0

        if state.fogger:

            fogger_gain = self.kf

        # -------------------------------------------------------------
        # Exhaust Ventilation
        # -------------------------------------------------------------

        ventilation_loss = 0.0

        if state.exhaust_fan:

            ventilation_loss = self.kv

        # -------------------------------------------------------------
        # Mushroom Transpiration
        # -------------------------------------------------------------

        transpiration_gain = (

            self.kt *

            state.fruit_weight

        )

        # -------------------------------------------------------------
        # Total Humidity Change
        # -------------------------------------------------------------

        delta_humidity = (

            fogger_gain

            -

            ventilation_loss

            +

            transpiration_gain

            -

            evaporation_loss

        )

        state.humidity += (

            Config.TIME_STEP *

            delta_humidity

        )

    # =========================================================================
    # CO₂ MODEL
    # =========================================================================

    def update_co2(
        self,
        state: TwinState
    ) -> None:
        """
        Chamber CO₂ Mass Balance

        CO₂(k+1) = CO₂(k)
                 + Respiration
                 - Ventilation
        """

        # -------------------------------------------------------------
        # Biological CO₂ Generation
        # -------------------------------------------------------------

        generation = (

            state.total_respiration

        )

        # -------------------------------------------------------------
        # Ventilation Removal
        # -------------------------------------------------------------

        ventilation = 0.0

        if state.exhaust_fan:

            ventilation = Config.CO2_VENTILATION

        elif state.fresh_air_fan:

            ventilation = (

                Config.CO2_VENTILATION * 0.60

            )

        # -------------------------------------------------------------
        # Chamber Mass Balance
        # -------------------------------------------------------------

        delta_co2 = (

            generation

            /

            Config.CHAMBER_VOLUME

            -

            ventilation

        )

        state.co2 += (

            Config.TIME_STEP *

            delta_co2

        )

    # =========================================================================
    # SUBSTRATE MOISTURE MODEL
    # =========================================================================

    def update_moisture(
        self,
        state: TwinState
    ) -> None:
        """
        Substrate Moisture Dynamics

        M(k+1)=M(k)

              + Irrigation

              - Evaporation

              - Biological Consumption
        """

        # -------------------------------------------------------------
        # Irrigation
        # -------------------------------------------------------------

        irrigation = 0.0

        if state.sprinkler:

            irrigation = self.ks

        # -------------------------------------------------------------
        # Evaporation
        # -------------------------------------------------------------

        evaporation = self.evaporation_rate(

            state.temperature,

            state.humidity

        )

        # -------------------------------------------------------------
        # Water Consumption by Mycelium
        # -------------------------------------------------------------

        biological_loss = (

            self.kg *

            state.colonization

        )

        # -------------------------------------------------------------
        # Moisture Balance
        # -------------------------------------------------------------

        delta_moisture = (

            irrigation

            -

            evaporation

            -

            biological_loss

        )

        state.moisture += (

            Config.TIME_STEP *

            delta_moisture

        )

    # =========================================================================
    # PHYSICAL LIMITS
    # =========================================================================

    def apply_limits(
        self,
        state: TwinState
    ) -> None:
        """
        Restrict all environmental variables
        to physically meaningful ranges.
        """

        state.temperature = self._clamp(

            state.temperature,

            0.0,

            50.0

        )

        state.humidity = self._clamp(

            state.humidity,

            0.0,

            100.0

        )

        state.co2 = self._clamp(

            state.co2,

            Config.BASELINE_CO2,

            Config.MAX_ALLOWED_CO2

        )

        state.moisture = self._clamp(

            state.moisture,

            0.0,

            100.0

        )

class GrowthModel:
    """
    Biological Growth Model

    Models
    ------
    1. Environmental Growth Modifier
    2. Logistic Colonization Model

    """

    def __init__(self):

        # -------------------------------------------------------------
        # Logistic Growth Parameters
        # -------------------------------------------------------------

        self.mu_max = Config.MU_MAX

        self.max_colonization = Config.MAX_COLONIZATION

        # -------------------------------------------------------------
        # Gompertz Parameter
        # -------------------------------------------------------------

        self.w_max = Config.W_MAX

        self.k = Config.GOMPERTZ_K

        self.t0 = Config.GOMPERTZ_T0

        self.pinning_threshold = Config.PINNING_THRESHOLD

    # ===============================================================
    # PRIVATE HELPER
    # ===============================================================

    def _linear_score(
        self,
        value: float,
        optimum_min: float,
        optimum_max: float,
        minimum: float,
        maximum: float
    ) -> float:
        """
        Returns a normalized score
        between 0 and 1.
        """

        if optimum_min <= value <= optimum_max:

            return 1.0

        if value < optimum_min:

            if value <= minimum:

                return 0.0

            return (

                (value - minimum)

                /

                (optimum_min - minimum)

            )

        if value >= maximum:

            return 0.0

        return (

            (maximum - value)

            /

            (maximum - optimum_max)

        )

    # ===============================================================
    # ENVIRONMENTAL MODIFIER
    # ===============================================================

    def environmental_modifier(
        self,
        state: TwinState
    ) -> float:
        """
        Calculates the effective growth modifier.

        μeff = μmax × fT × fH × fCO₂ × fM
        """

        # -------------------------------------------------------------
        # Temperature Modifier
        # -------------------------------------------------------------

        f_temperature = self._linear_score(

            state.temperature,

            Config.MIN_TEMPERATURE,

            Config.MAX_TEMPERATURE,

            15.0,

            40.0

        )

        # -------------------------------------------------------------
        # Humidity Modifier
        # -------------------------------------------------------------

        f_humidity = self._linear_score(

            state.humidity,

            Config.MIN_HUMIDITY,

            Config.MAX_HUMIDITY,

            50.0,

            100.0

        )

        # -------------------------------------------------------------
        # CO₂ Modifier
        # -------------------------------------------------------------

        f_co2 = self._linear_score(

            state.co2,

            Config.MIN_CO2,

            Config.MAX_CO2,

            400.0,

            5000.0

        )

        # -------------------------------------------------------------
        # Moisture Modifier
        # -------------------------------------------------------------

        f_moisture = self._linear_score(

            state.moisture,

            Config.MIN_MOISTURE,

            Config.MAX_MOISTURE,

            30.0,

            100.0

        )

        modifier = (

            f_temperature *

            f_humidity *

            f_co2 *

            f_moisture

        )

        modifier = max(

            0.0,

            min(

                modifier,

                1.0

            )

        )

        return modifier

    # ===============================================================
    # LOGISTIC COLONIZATION MODEL
    # ===============================================================

    def update_colonization(
        self,
        state: TwinState
    ) -> None:
        """
        Logistic Colonization Model

        dC/dt = μeff C (1 - C/K)
        """

        modifier = self.environmental_modifier(

            state

        )

        mu_effective = (

            self.mu_max *

            modifier

        )

        growth_rate = (

            mu_effective *

            state.colonization *

            (

                1.0 -

                (

                    state.colonization

                    /

                    self.max_colonization

                )

            )

        )

        state.colonization += (

            Config.TIME_STEP *

            growth_rate

        )

        if state.colonization > self.max_colonization:

            state.colonization = self.max_colonization

        state.colonization_percentage = (

            state.colonization

        )

    # ===============================================================
    # GOMPERTZ FRUIT GROWTH MODEL
    # ===============================================================

    def update_fruit_growth(
        self,
        state: TwinState
    ) -> None:
        """
        Gompertz Fruit Growth Model

        W(t) = Wmax * exp[-exp(-k(t-t0))]

        Fruit growth starts only after the
        pinning threshold is reached.
        """

        # -------------------------------------------------------------
        # Fruiting starts only after sufficient colonization
        # -------------------------------------------------------------

        if state.colonization < self.pinning_threshold:

            state.fruit_weight = 0.0

            state.pinhead_count = 0

            state.fruit_count = 0

            return

        # -------------------------------------------------------------
        # Simulation Time (Days)
        # -------------------------------------------------------------

        simulation_day = (

            state.current_hour

            /

            24.0

        )

        # -------------------------------------------------------------
        # Gompertz Growth
        # -------------------------------------------------------------

        fruit_weight = (

            self.w_max

            *

            math.exp(

                -

                math.exp(

                    -

                    self.k

                    *

                    (

                        simulation_day

                        -

                        self.t0

                    )

                )

            )

        )

        # -------------------------------------------------------------
        # Prevent Numerical Errors
        # -------------------------------------------------------------

        fruit_weight = max(

            0.0,

            fruit_weight

        )

        state.fruit_weight = fruit_weight

        # -------------------------------------------------------------
        # Approximate Pinhead Count
        # -------------------------------------------------------------

        state.pinhead_count = int(

            max(

                1,

                fruit_weight / 12.0

            )

        )

        # -------------------------------------------------------------
        # Approximate Fruit Count
        # -------------------------------------------------------------

        state.fruit_count = int(

            max(

                1,

                fruit_weight / 35.0

            )

        )

    # ===============================================================
    # MAIN UPDATE
    # ===============================================================

    def update(
        self,
        state: TwinState
    ) -> None:
        """
        Executes one biological growth step.
        """

        self.update_colonization(

            state

        )

        self.update_fruit_growth(

            state

        )

        self.estimate_yield(

            state

        )

    # VALIDATION
    # ===============================================================

    def validate(
        self,

    ) -> bool:
        """
        Validate biological variables.
        """

        if state.colonization < 0.0:

class RespirationModel:
    """
    Biological Respiration Model

    Models
    ------
    1. Substrate Respiration
    2. Fruit Respiration.
    """

    def __init__(self):

        # -------------------------------------------------------------
        # Jung & Son (2021)
        # -------------------------------------------------------------

        self.a1 = Config.A1
        self.a2 = Config.A2
        self.a3 = Config.A3

        # -------------------------------------------------------------
        # Thornley Model
        # -------------------------------------------------------------

        self.maintenance = Config.MAINTENANCE_COEFFICIENT
        self.growth = Config.GROWTH_COEFFICIENT

        # -------------------------------------------------------------
        # Heat Conversion
        # -------------------------------------------------------------

        self.heat_coefficient = Config.RESPIRATION_HEAT_COEFFICIENT

        # -------------------------------------------------------------
        # Previous Fruit Weight
        # -------------------------------------------------------------

        self.previous_fruit_weight = 0.0

    # ===============================================================
    # SUBSTRATE RESPIRATION
    # ===============================================================

    def substrate_respiration(
        self,
        state: TwinState
    ) -> float:
        """
        Jung & Son (2021)

        Rm = a1 * exp(a2*T) + a3
        """

        respiration = (

            self.a1 *

            math.exp(

                self.a2 *

                state.temperature

            )

            +

            self.a3

        )

        respiration = max(

            0.0,

            respiration

        )

        state.substrate_respiration = respiration

        return respiration

    # ===============================================================
    # FRUIT RESPIRATION
    # ===============================================================

    def fruit_respiration(
        self,
        state: TwinState
    ) -> float:
        """
        Thornley Respiration Model

        R = MW + P(dW/dt)
        """

        # -------------------------------------------------------------
        # Fruit Growth Rate
        # -------------------------------------------------------------

        growth_rate = (

            state.fruit_weight

            -

            self.previous_fruit_weight

        )

        # -------------------------------------------------------------
        # Respiration
        # -------------------------------------------------------------

        respiration = (

            self.maintenance *

            state.fruit_weight

            +

            self.growth *

            growth_rate

        )

        respiration = max(

            0.0,

            respiration

        )

        self.previous_fruit_weight = (

            state.fruit_weight

        )

        state.fruit_respiration = respiration

        return respiration

    # ===============================================================
    # RESPIRATION STATUS
    # ===============================================================

    def get_respiration_status(
        self,
        state: TwinState
    ) -> dict:

        return {

            "Substrate Respiration":

                state.substrate_respiration,

            "Fruit Respiration":

                state.fruit_respiration

        }

class HealthModel:
    """
    Digital Twin Health Assessment Model

    Computes

    1. Environmental Health Score (EHS)
    2. Growth Health Score (GHS)
    3. Contamination Health Score (CHS)
    """

    def __init__(self):

        self.environment_weight = Config.ECS_WEIGHT

        self.growth_weight = Config.GPS_WEIGHT

        self.contamination_weight = Config.CCS_WEIGHT

    # ===============================================================
    # PRIVATE HELPER
    # ===============================================================

    def _score(
        self,
        value: float,
        optimum_min: float,
        optimum_max: float,
        minimum: float,
        maximum: float
    ) -> float:
        """
        Returns a normalized score
        between 0 and 100.
        """

        if optimum_min <= value <= optimum_max:

            return 100.0

        if value < optimum_min:

            if value <= minimum:

                return 0.0

            score = (

                (value - minimum)

                /

                (optimum_min - minimum)

            )

            return score * 100.0

        if value >= maximum:

            return 0.0

        score = (

            (maximum - value)

            /

            (maximum - optimum_max)

        )

        return score * 100.0

    # ===============================================================
    # ENVIRONMENT HEALTH SCORE
    # ===============================================================

    def environment_score(
        self,
        state: TwinState
    ) -> float:
        """
        Calculates environmental health.
        """

        temperature_score = self._score(

            state.temperature,

            Config.MIN_TEMPERATURE,

            Config.MAX_TEMPERATURE,

            15.0,

            40.0

        )

        humidity_score = self._score(

            state.humidity,

            Config.MIN_HUMIDITY,

            Config.MAX_HUMIDITY,

            50.0,

            100.0

        )

        co2_score = self._score(

            state.co2,

            Config.MIN_CO2,

            Config.MAX_CO2,

            400.0,

            Config.MAX_ALLOWED_CO2

        )

        moisture_score = self._score(

            state.moisture,

            Config.MIN_MOISTURE,

            Config.MAX_MOISTURE,

            30.0,

            100.0

        )

        environment = (

            temperature_score

            +

            humidity_score

            +

            co2_score

            +

            moisture_score

        ) / 4.0

        state.environment_score = environment

        return environment

    # ===============================================================
    # GROWTH HEALTH SCORE
    # ===============================================================

    def growth_score(
        self,
        state: TwinState
    ) -> float:
        """
        Growth health based on colonization.
        """

        score = max(

            0.0,

            min(

                state.colonization,

                100.0

            )

        )

        state.growth_performance_score = score

        return score

    # ===============================================================
    # CONTAMINATION HEALTH SCORE
    # ===============================================================

    def contamination_score(
        self,
        state: TwinState
    ) -> float:
        """
        Computes contamination score.
        """

        contamination = max(

            0.0,

            min(

                state.contamination_level,

                100.0

            )

        )

        score = 100.0 - contamination

        state.contamination_score = score

        return score

class StageModel:
    """
    Growth Stage Model

    Determines the current cultivation stage
    based on the biological state of the crop.

    Stages

    1. Preparation
    2. Inoculation
    3. Spawn Run
    4. Pinhead Formation
    5. Fruiting
    6. Harvesting
    """

    def __init__(self):

        self.stages = [

            "Preparation",

            "Inoculation",

            "Spawn Run",

            "Pinhead Formation",

            "Fruiting",

            "Harvesting"

        ]

    # ===============================================================
    # PREPARATION
    # ===============================================================

    def is_preparation(
        self,
        state: TwinState
    ) -> bool:

        return state.current_day == 0

    # ===============================================================
    # INOCULATION
    # ===============================================================

    def is_inoculation(
        self,
        state: TwinState
    ) -> bool:

        return (

            state.current_day >= 1

            and

            state.colonization < 5

        )

    # ===============================================================
    # SPAWN RUN
    # ===============================================================

    def is_spawn_run(
        self,
        state: TwinState
    ) -> bool:

        return (

            5

            <=

            state.colonization

            <

            Config.PINNING_THRESHOLD

        )

    # ===============================================================
    # PINHEAD FORMATION
    # ===============================================================

    def is_pinhead(
        self,
        state: TwinState
    ) -> bool:

        return (

            state.colonization

            >=

            Config.PINNING_THRESHOLD

            and

            state.pinhead_count > 0

            and

            state.fruit_weight < 50

        )

    # ===============================================================
    # FRUITING
    # ===============================================================

    def is_fruiting(
        self,
        state: TwinState
    ) -> bool:

        return (

            state.fruit_weight >= 50

            and

            state.fruit_weight <

    # HARVESTING

class DigitalTwin:
    """
    Scientific Digital Twin Engine

    Integrates

    • Environment Model
    • Growth Model
    • Respiration Model
    • Health Model
    • Stage Model
    • Data Logger
    • Plotter
    """

    def __init__(self):

        # -------------------------------------------------------------
        # Core State
        # -------------------------------------------------------------

        self.state = TwinState()

        # -------------------------------------------------------------
        # Scientific Models
        # -------------------------------------------------------------

        self.environment = EnvironmentModel()

        self.growth = GrowthModel()

        self.respiration = RespirationModel()

        self.health = HealthModel()

        self.stage = StageModel()

        # -------------------------------------------------------------
        # Utilities
        # -------------------------------------------------------------

        self.logger = DataLogger()

        self.plotter = Plotter()

    # ===============================================================
    # ONE SIMULATION STEP
    # ===============================================================

    def step(self) -> None:
        """
        Execute one simulation step.
        """

        # -------------------------------------------------------------
        # Biological Growth
        # -------------------------------------------------------------

        self.growth.update(

            self.state

        )

        # -------------------------------------------------------------
        # Respiration
        # -------------------------------------------------------------

        self.respiration.update(

            self.state

        )

        # -------------------------------------------------------------
        # Environment
        # -------------------------------------------------------------

        self.environment.update(

            self.state

        )

        # -------------------------------------------------------------
        # Health Assessment
        # -------------------------------------------------------------

        self.health.update(

            self.state

        )

        # -------------------------------------------------------------
        # Stage Identification
        # -------------------------------------------------------------

        self.stage.update(

            self.state

        )

        # -------------------------------------------------------------
        # Data Logging
        # -------------------------------------------------------------

        self.logger.log(

            self.state

        )

        # -------------------------------------------------------------
        # Advance Simulation Time
        # -------------------------------------------------------------

        self.state.update_time()

    # ===============================================================
    # MULTIPLE STEPS
    # ===============================================================

    def simulate(
        self,
        hours: int
    ) -> None:
        """
        Run simulation for a specified
        number of hours.
        """

        for _ in range(hours):

            self.step()

    # ===============================================================
    # RESET
    # ===============================================================

    def reset(self) -> None:

        self.state = TwinState()

        self.logger.clear()

    # ===============================================================
    # CURRENT STATUS
    # ===============================================================

    def status(self) -> None:

        print("\n")

        print("=" * 70)

        print("DIGITAL TWIN STATUS")

        print("=" * 70)

        print(

            f"Hour           : {self.state.current_hour}"

        )

        print(

            f"Day            : {self.state.current_day}"

        )

        print(

            f"Stage          : {self.state.current_stage}"

        )

        print(

            f"Temperature    : {self.state.temperature:.2f} °C"

        )

        print(

            f"Humidity       : {self.state.humidity:.2f} %"

        )

        print(

            f"CO₂            : {self.state.co2:.2f} ppm"

        )

        print(

            f"Moisture       : {self.state.moisture:.2f} %"

        )

        print(

            f"Colonization   : {self.state.colonization:.2f} %"

        )

        print(

            f"Fruit Weight   : {self.state.fruit_weight:.2f} g"

        )

        print(

            f"Yield          : {self.state.yield_estimate:.2f} g"

        )

        print(

            f"Health Index   : {self.state.health_index:.2f}"

        )

        print("=" * 70)

    # ===============================================================
    # PLACEHOLDERS
    # ===============================================================

    def export_results(
        self,
        filename="simulation_results.csv"
    ):

        pass

    def visualize(self):

        pass

    def validate(self):

        pass


    # ===============================================================
    # EXPORT RESULTS
    # ===============================================================

    def export_results(
        self,
        filename: str = "simulation_results.csv"
    ) -> None:
        """
        Export simulation results to CSV.
        """

        self.logger.export_csv(

            filename

        )

    # ===============================================================
    # VISUALIZATION
    # ===============================================================

    def visualize(self) -> None:
        """
        Generate all simulation plots.
        """

        self.plotter.plot_all(

            self.logger

        )

    # ===============================================================
    # COMPLETE REPORT
    # ===============================================================

    def report(self) -> None:
        """
        Print complete digital twin report.
        """

        print("\n")

        print("=" * 70)

        print("SCIENTIFIC DIGITAL TWIN REPORT")

        print("=" * 70)

        self.status()

        self.logger.summary()

        print("=" * 70)

    # ===============================================================
    # VALIDATION
    # ===============================================================

    def validate(self) -> bool:
        """
        Validate all digital twin models.
        """

        validation_results = [

            self.environment.validate(

                self.state

            ),

            self.growth.validate(

                self.state

            ),

            self.respiration.validate(

                self.state

            ),

            self.health.validate(

                self.state

            ),

            self.stage.validate(

                self.state

            ),

            self.logger.validate()

        ]

        return all(

            validation_results

        )

    # ===============================================================
    # RUN COMPLETE SIMULATION
    # ===============================================================

    def run(
        self,
        days: int = Config.SIMULATION_DAYS
    ) -> None:
        """
        Execute complete simulation.
        """

        total_hours = (

            days *

            Config.HOURS_PER_DAY

        )

        print("\n")

        print("=" * 70)

        print("STARTING DIGITAL TWIN SIMULATION")

        print("=" * 70)

        self.simulate(

            total_hours

        )

        print(

            "\nSimulation Completed Successfully."

        )

        print(

            f"Simulation Time : {days} days"

        )

        print(

            f"Total Steps     : {total_hours}"

        )

        print("=" * 70)

    # ===============================================================
    # FINAL STATE
    # ===============================================================

    def final_state(
        self
    ) -> TwinState:
        """
        Return the final twin state.
        """

        return self.state

    # ===============================================================
    # GET DATAFRAME
    # ===============================================================

    def dataframe(
        self
    ) -> pd.DataFrame:
        """
        Return simulation results as DataFrame.
        """

        return self.logger.to_dataframe()

class MasterFSM:
    """
    Master Finite State Machine

    Supervises all subsystem FSMs.
    """

    def __init__(self):

        self.environment_state = EnvironmentState.NORMAL

        self.temperature_state = TemperatureState.NORMAL

        self.humidity_state = HumidityState.NORMAL

        self.co2_state = CO2State.NORMAL

        self.moisture_state = MoistureState.NORMAL

        self.contamination_state = ContaminationState.SAFE

        self.equipment_state = EquipmentState.NORMAL

    # ===============================================================
    # TEMPERATURE FSM
    # ===============================================================

    def update_temperature(
        self,
        state: TwinState
    ) -> None:

        if state.temperature < Config.MIN_TEMPERATURE:

            self.temperature_state = TemperatureState.LOW

        elif state.temperature > 33.0:

            self.temperature_state = TemperatureState.CRITICAL

        elif state.temperature > Config.MAX_TEMPERATURE:

            self.temperature_state = TemperatureState.HIGH

        else:

            self.temperature_state = TemperatureState.NORMAL

    # ===============================================================
    # HUMIDITY FSM
    # ===============================================================

    def update_humidity(
        self,
        state: TwinState
    ) -> None:

        if state.humidity < Config.MIN_HUMIDITY:

            self.humidity_state = HumidityState.LOW

        elif state.humidity > 95.0:

            self.humidity_state = HumidityState.CRITICAL

        elif state.humidity > Config.MAX_HUMIDITY:

            self.humidity_state = HumidityState.HIGH

        else:

            self.humidity_state = HumidityState.NORMAL

    # ===============================================================
    # CO₂ FSM
    # ===============================================================

    def update_co2(
        self,
        state: TwinState
    ) -> None:

        if state.co2 < Config.MIN_CO2:

            self.co2_state = CO2State.LOW

        elif state.co2 > 2000:

            self.co2_state = CO2State.CRITICAL

        elif state.co2 > Config.MAX_CO2:

            self.co2_state = CO2State.HIGH

        else:

            self.co2_state = CO2State.NORMAL

    # ===============================================================
    # MOISTURE FSM
    # ===============================================================

    def update_moisture(
        self,
        state: TwinState
    ) -> None:

        if state.moisture < Config.MIN_MOISTURE:

            self.moisture_state = MoistureState.LOW

        elif state.moisture > Config.MAX_MOISTURE:

            self.moisture_state = MoistureState.HIGH

        else:

            self.moisture_state = MoistureState.NORMAL

    # ===============================================================
    # CONTAMINATION FSM
    # ===============================================================

    def update_contamination(
        self,
        state: TwinState
    ) -> None:

        contamination = state.contamination_level

        if contamination >= 30:

            self.contamination_state = ContaminationState.EMERGENCY

        elif contamination >= 10:

            self.contamination_state = ContaminationState.CONTAMINATED

        elif contamination > 0:

            self.contamination_state = ContaminationState.WARNING

        else:

            self.contamination_state = ContaminationState.SAFE

# =============================================================================
# MASTER FSM (CONTINUED)
# =============================================================================

    # ===============================================================
    # ENVIRONMENT FSM
    # ===============================================================

    def update_environment(
        self
    ) -> None:
        """
        Determines the overall environmental state
        from all subsystem FSMs.
        """

        if (

            self.temperature_state == TemperatureState.CRITICAL

            or

            self.humidity_state == HumidityState.CRITICAL

            or

            self.co2_state == CO2State.CRITICAL

            or

            self.contamination_state == ContaminationState.EMERGENCY

        ):

            self.environment_state = EnvironmentState.EMERGENCY

            return

        if (

            self.temperature_state == TemperatureState.HIGH

            or

            self.humidity_state == HumidityState.HIGH

            or

            self.co2_state == CO2State.HIGH

            or

            self.contamination_state == ContaminationState.CONTAMINATED

        ):

            self.environment_state = EnvironmentState.CRITICAL

            return

        if (

            self.temperature_state == TemperatureState.LOW

            or

            self.humidity_state == HumidityState.LOW

            or

            self.co2_state == CO2State.LOW

            or

            self.moisture_state == MoistureState.LOW

            or

            self.contamination_state == ContaminationState.WARNING

        ):

            self.environment_state = EnvironmentState.WARNING

            return

        self.environment_state = EnvironmentState.NORMAL

    # ===============================================================
    # EQUIPMENT FSM
    # ===============================================================

    def update_equipment(
        self,
        state: TwinState
    ) -> None:
        """
        Determines whether any actuator is active.
        """

        active = (

            state.cooling_pad

            or

            state.heater

            or

            state.fogger

            or

            state.sprinkler

            or

            state.exhaust_fan

            or

            state.fresh_air_fan

        )

        if active:

            self.equipment_state = EquipmentState.ACTIVE

        else:

            self.equipment_state = EquipmentState.NORMAL

    # ===============================================================
    # MASTER UPDATE
    # ===============================================================

    def update(
        self,
        state: TwinState
    ) -> None:
        """
        Executes one complete FSM update.
        """

        self.update_temperature(

            state

        )

        self.update_humidity(

            state

        )

        self.update_co2(

            state

        )

        self.update_moisture(

            state

        )

        self.update_contamination(

            state

        )

        self.update_environment()

        self.update_equipment(

            state

        )

class ActuatorController:
    """
    Intelligent Actuator Controller

    Executes automatic control actions
    based on FSM states.

    Controlled Devices
    ------------------
    • Cooling Pad
    • Heater
    • Fogger
    • Sprinkler
    • Exhaust Fan
    • Fresh Air Fan
    """

    def __init__(self):

        pass

    # ===============================================================
    # RESET ALL ACTUATORS
    # ===============================================================

    def reset(
        self,
        state: TwinState
    ) -> None:

        state.cooling_pad = False

        state.heater = False

        state.fogger = False

        state.sprinkler = False

        state.exhaust_fan = False

        state.fresh_air_fan = False

    # ===============================================================
    # TEMPERATURE CONTROL
    # ===============================================================

    def temperature_control(
        self,
        state: TwinState,
        fsm: MasterFSM
    ) -> None:

        if fsm.temperature_state == TemperatureState.LOW:

            state.heater = True

        elif fsm.temperature_state == TemperatureState.HIGH:

            state.cooling_pad = True

            state.exhaust_fan = True

        elif fsm.temperature_state == TemperatureState.CRITICAL:

            state.cooling_pad = True

            state.exhaust_fan = True

        else:

            state.heater = False

            state.cooling_pad = False

    # ===============================================================
    # HUMIDITY CONTROL
    # ===============================================================

    def humidity_control(
        self,
        state: TwinState,
        fsm: MasterFSM
    ) -> None:

        if fsm.humidity_state == HumidityState.LOW:

            state.fogger = True

        elif fsm.humidity_state == HumidityState.HIGH:

            state.fogger = False

            state.exhaust_fan = True

        elif fsm.humidity_state == HumidityState.CRITICAL:

            state.fogger = False

            state.exhaust_fan = True

            state.cooling_pad = False

        else:

            state.fogger = False

    # ===============================================================
    # CO₂ CONTROL
    # ===============================================================

    def co2_control(
        self,
        state: TwinState,
        fsm: MasterFSM
    ) -> None:

        if fsm.co2_state == CO2State.HIGH:

            state.fresh_air_fan = True

            state.exhaust_fan = True

        elif fsm.co2_state == CO2State.CRITICAL:

            state.fresh_air_fan = True

            state.exhaust_fan = True

        elif fsm.co2_state == CO2State.LOW:

            state.fresh_air_fan = False

        else:

            state.fresh_air_fan = False

    # ===============================================================
    # MOISTURE CONTROL
    # ===============================================================

    def moisture_control(
        self,
        state: TwinState,
        fsm: MasterFSM
    ) -> None:

        if fsm.moisture_state == MoistureState.LOW:

            state.sprinkler = True

        elif fsm.moisture_state == MoistureState.HIGH:

            state.sprinkler = False

            state.exhaust_fan = True

        else:

            state.sprinkler = False

    # ===============================================================
    # PLACEHOLDERS
    # ===============================================================

    def contamination_control(
        self,
        state: TwinState,
        fsm: MasterFSM
    ):

        pass

    def update(
        self,
        state: TwinState,
        fsm: MasterFSM
    ):

        pass

    def status(
        self,
        state: TwinState
    ):

        pass


    # ===============================================================
    # CONTAMINATION CONTROL
    # ===============================================================

    def contamination_control(
        self,
        state: TwinState,
        fsm: MasterFSM
    ) -> None:
        """
        Control actions for contamination events.
        """

        if fsm.contamination_state == ContaminationState.WARNING:

            state.exhaust_fan = True

        elif fsm.contamination_state == ContaminationState.CONTAMINATED:

            state.sprinkler = False

            state.fogger = False

            state.exhaust_fan = True

        elif fsm.contamination_state == ContaminationState.EMERGENCY:

            state.sprinkler = False

            state.fogger = False

            state.cooling_pad = False

            state.exhaust_fan = True

            state.fresh_air_fan = False

    # ===============================================================
    # MASTER CONTROLLER
    # ===============================================================

    def update(
        self,
        state: TwinState,
        fsm: MasterFSM
    ) -> None:
        """
        Execute complete actuator control.
        """

        # ----------------------------------------------------------
        # Reset Actuators
        # ----------------------------------------------------------

        self.reset(

            state

        )

        # ----------------------------------------------------------
        # Environmental Controllers
        # ----------------------------------------------------------

        self.temperature_control(

            state,

            fsm

        )

        self.humidity_control(

            state,

            fsm

        )

        self.co2_control(

            state,

            fsm

        )

        self.moisture_control(

            state,

            fsm

        )

        self.contamination_control(

            state,

            fsm

        )

    # ===============================================================
    # CONTROLLER STATUS
    # ===============================================================

    def get_status(
        self,
        state: TwinState
    ) -> dict:
        """
        Returns actuator states.
        """

        return {

            "Cooling Pad":

                state.cooling_pad,

            "Heater":

                state.heater,

            "Fogger":

                state.fogger,

            "Sprinkler":

                state.sprinkler,

            "Exhaust Fan":

                state.exhaust_fan,

            "Fresh Air Fan":

                state.fresh_air_fan

        }

    # ===============================================================
    # PRINT STATUS
    # ===============================================================

    def print_status(
        self,
        state: TwinState
    ) -> None:

        print("\n")

        print("=" * 70)

        print("ACTUATOR STATUS")

        print("=" * 70)

        print(

            f"Cooling Pad    : {state.cooling_pad}"

        )

        print(

            f"Heater         : {state.heater}"

        )

        print(

            f"Fogger         : {state.fogger}"

        )

        print(

            f"Sprinkler      : {state.sprinkler}"

        )

        print(

            f"Exhaust Fan    : {state.exhaust_fan}"

        )

        print(

            f"Fresh Air Fan  : {state.fresh_air_fan}"

        )

        print("=" * 70)

    # ===============================================================
    # VALIDATION
    # ===============================================================

    def validate(
        self,
        state: TwinState
    ) -> bool:
        """
        Validate actuator variables.
        """

        actuator_states = [

            state.cooling_pad,

            state.heater,

            state.fogger,

            state.sprinkler,

            state.exhaust_fan,

            state.fresh_air_fan

        ]

        return all(

            isinstance(

                actuator,

                bool

            )

            for actuator in actuator_states

        )

class IntelligentDigitalTwin(DigitalTwin):
    """
    Complete Scientific Digital Twin

    Integrated Modules

    • Environment Model
    • Growth Model
    • Respiration Model
    • Health Model
    • Stage Model
    • Master FSM
    • Actuator Controller
    • Alert Manager
    • Computer Vision Interface
    • ESP32 Interface
    • Dashboard Interface
    • Data Logger
    """

    def __init__(self):

        super().__init__()

        # ------------------------------------------------------------
        # Intelligence Layer
        # ------------------------------------------------------------

        self.fsm = MasterFSM()

        self.controller = ActuatorController()

        self.alerts = AlertManager()

        # ------------------------------------------------------------
        # Interfaces
        # ------------------------------------------------------------

        self.vision = ComputerVisionInterface()

        self.esp32 = ESP32Interface()

        self.dashboard = DashboardInterface()

    # ===============================================================
    # COMPLETE DIGITAL TWIN CYCLE
    # ===============================================================

    def step(self) -> None:
        """
        Executes one complete Digital Twin cycle.
        """

        # ------------------------------------------------------------
        # Read Physical Sensors
        # ------------------------------------------------------------

        self.esp32.update_twin(

            self.state

        )

        # ------------------------------------------------------------
        # Computer Vision
        # ------------------------------------------------------------

        self.vision.update_state(

            self.state

        )

        # ------------------------------------------------------------
        # Scientific Models
        # ------------------------------------------------------------

        self.growth.update(

            self.state

        )

        self.respiration.update(

            self.state

        )

        self.environment.update(

            self.state

        )

        self.health.update(

            self.state

        )

        self.stage.update(

            self.state

        )

        # ------------------------------------------------------------
        # FSM Decision Layer
        # ------------------------------------------------------------

        self.fsm.update(

            self.state

        )

        # ------------------------------------------------------------
        # Intelligent Controller
        # ------------------------------------------------------------

        self.controller.update(

            self.state,

            self.fsm

        )

        # ------------------------------------------------------------
        # Alerts
        # ------------------------------------------------------------

        self.alerts.update(

            self.fsm

        )

        # ------------------------------------------------------------
        # Dashboard
        # ------------------------------------------------------------

        self.dashboard.update(

            self.state

        )

        # ------------------------------------------------------------
        # Data Logger
        # ------------------------------------------------------------

        self.logger.log(

            self.state

        )

        # ------------------------------------------------------------
        # Advance Time
        # ------------------------------------------------------------

        self.state.update_time()

    # ===============================================================
    # RUN COMPLETE SYSTEM
    # ===============================================================

    def run_system(
        self,
        days: int = Config.SIMULATION_DAYS
    ) -> None:

        total_steps = (

            days *

            Config.HOURS_PER_DAY

        )

        print("\n")

        print("=" * 80)

        print("SCIENTIFIC DIGITAL TWIN INITIALIZED")

        print("=" * 80)

        print(f"Simulation Duration : {days} Days")

        print(f"Total Steps         : {total_steps}")

        print("=" * 80)

        for _ in range(total_steps):

            self.step()

        print("\nSimulation Completed Successfully.")

    # ===============================================================
    # COMPLETE SYSTEM REPORT
    # ===============================================================

    def complete_report(
        self
    ) -> None:

        self.status()

        self.fsm.print_status()

        self.controller.print_status(

            self.state

        )

        self.health.print_status(

            self.state

        )

        self.stage.print_status(

            self.state

        )

        self.logger.summary()

        self.alerts.print_alerts()

        self.dashboard.summary()

    # ===============================================================
    # EXPORT EVERYTHING
    # ===============================================================

    def export_all(
        self
    ) -> None:

        self.export_results(

            "simulation_results.csv"

        )

        self.dashboard.export_json(

            "dashboard.json"

        )

    # ===============================================================
    # VALIDATE COMPLETE SYSTEM
    # ===============================================================

    def validate_system(
        self
    ) -> bool:

        return all([

            self.environment.validate(

                self.state

            ),

            self.growth.validate(

                self.state

            ),

            self.respiration.validate(

                self.state

            ),

            self.health.validate(

                self.state

            ),

            self.stage.validate(

                self.state

            ),

            self.logger.validate(),

            self.fsm is not None,

            self.controller.validate(

                self.state

            ),

            self.alerts.validate(),

            self.vision.validate(),

            self.esp32.validate(),

            self.dashboard.validate()

        ])


# =============================================================================
# MAIN PROGRAM
# =============================================================================

def main():

    twin = IntelligentDigitalTwin()

    if twin.validate_system():

        twin.run_system()

        twin.complete_report()

        twin.export_all()

        twin.visualize()

    else:

        print("\nSystem Validation Failed.")


if __name__ == "__main__":

    main()
